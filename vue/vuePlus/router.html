# 前端路由的实现

  1. BaseRouter
const ELEMENT = document.querySelector('#page')
class BaseRouter {
  constructor(list) {
    this.list = list
  }
  // 渲染页面
  render(state) {
    let ele = this.list.find(ele => ele.path === state)
    ele = ele ? ele : this.find(ele => ele.path === '*')
    ELEMENT.innerText = ele.component
  }
}

  1. HashRouter
export HashRouter extends BaseRouter{
  constructor(){
     super(list)
    //  监听hashChange方法
    this.handle()
    window.addEventListener('hashchange',function(){
      this.handle()
    })
  },
  // 调用render方法
  handle(){
    this.render(this.getState())
  }
  // 获取hash
  getState(){
    const hash = window.location.hash
     return hash ? hash : '/'
  },
  getUrl(path){
     const href = window.location.href
     const i = href.include('#')
     const base = i > 0 ? href.slice(0, i) : href
     return `${base}#${path}`
  }
  // push方法
  push(path){
    window.location.hash = path
  },
  go(n){
    window.location.go(n)
  },
  replace(path){
    window.location.replace(this.getUrl(path))
  }
}

  1. HistoryRouter
export class HistoryRouter extends BaseRouter {
  constructor() {
    super(list)
    this.handle()
    window.addEventListener('popstate', () => {
      this.handle()
    })
  },
  handle(){
    this.render(this.getState(path))
  }
  getState(path){
      const path = window.location.pathname
      return path ? path : '/'
  }
  push(path){
      history.pushState(null, null, path)
      this.handle()
  }
   replace(path) {
    history.replaceState(null, null, path);
    this.handler();
  }
  go(n){
     window.history.go(n)
  }
}

  1. webRouter
import { HashRouter } from './hash'
import { HistoryRouter } from './history'
import { ROUTELIST } from './routeList'

const MODE = 'history'

class webRouter {
  constructor({ mode = 'hash', routeList }) {
    this.router =
      mode === 'hash' ? new HashRouter(routeList) : new HistoryRouter(routeList)
  }
  push(path) {
    this.router.push(path)
  }
  go(n) {
    this.router.go(n)
  }
  replace(path) {
    this.router.replace(path)
  }
}

const router = new webRouter({
  mode: MODE,
  routeList: ROUTELIST
})
