# 匿名字段

type Person struct{
  name   string
  sex    string
  age    int
}

type Student struct{
  Person
  id     int
  score  int
}

func main () {
  // 第一种方式
  var stu Student
  stu.id = 101
  stu.name = "张三"  // 或者 stu.Person.name = "张三"
  stu.score = 100
  stu。sex ="男"
  stu.age =18

// 第二种方式
var stu Student = Student{Person{name:"张三",sex:"男",age:18},id:101,score:90}
}

# 同名字段

type Person struct{
  name   string
  sex    string
  age    int
}

type Student struct{
  Person
  id     int
  score  int
  name   string
}

func main () {
    // 第一种方式
  var stu Student
  stu.id = 101
  // 采用就近原则
  stu.name = "张三"
  stu.Person.name = "张三疯"
  stu.score = 100
  stu。sex ="男"
  stu.age =18

  // 第二种方式
var stu Student = Student{Person{name:"张三",sex:"男",age:18},id:101,score:90,name:"张三疯"}
}

# 指针匿名字段


type person struct{
  name   string
  age    int
}

type student struct{
  *person
  id     int
  score   int
}

func main () {
  var stu student
  // error，因为这个时候指针是一个空指针
  // stu.name = "郭襄"

  // 第一种做法
  stu.person = new(person)
  stu.name = "郭襄"


  // 第二种做法
  var per person = person{name:"杨过",age:18}
  stu.person = &per
  stu.name = "小龙女"
}

# 方法的定义和使用 02

type Stu struct{
  name    string
  age      int
}

func (s Stu)printInfo(){
  fmt.printIn(s.name)
  fmt.printIn(s.age)
}

func (s *Stu)editInfo(name string,age int){
  s.name = name
  s.age = age
}

func main (){
  var s Stu = Stu{name:"刘姥姥",age:54}
  s.printInfo()

  (&s).editInfo(name:"林黛玉",age:16)
}
