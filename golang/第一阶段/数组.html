# 冒泡排序

func main(){
  arr:=[6]int {4,8,4,1,9,2}
  for i:=0;i<len(arr)-1;i++{
    for j:=0;j<len(arr)-1-i;j++{
      if arr[j]>arr[j+1]{
        arr[j],arr[j+1]=arr[j+1],arr[j]
      }
    }
  }
}

# 随机数

  3. 添加随机数
1.
func main(){
  for i:=0;i<10;i++{
    fmt.printIn(rand.Intn(123))
    //这个时候会得到相同的值，因为没有导入时间
  }
}

func main(){
  // 随机数种子
  rand.Seed(time.Now().UnixNano())
  for i:=0;i<10;i++{
    fmt.printIn(rand.Intn(100))  // 0-99 的数字

  }
}

案例,像福利裁片一样，随机生成 6 个红球（1-33），不重复，加上一个蓝球（1-16）

func main(){
  rand.Seed(time.Now().UnixNano())
  var arr [6]int

  for i:=0;i<len(arr);i++{
    temp := rand.Intn(33)+1
    for j:=0;j<i;j++{
      if temp == arr[j]{
        temp := rand.Intn(33)+1
        j=-1
        continue
      }
    }
    arr[i] = temp
  }
}
