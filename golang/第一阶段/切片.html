# 切片是什么？

其实切片相当于其他语言中的动态数组，在 go 语言中，数组的长度不能发生改变，而切片就是可以改变长度的数组

# 切片的定义

func main () {


**********第一种定义方法*************
  var slice []int = []int{1,2,3,4}
************************************

**********第二种定义方法*************
  slice := []int{1,2,3,4}  // 自动类型推导
************************************

**********第三种定义方法*************
  slice := make([]int,1)
************************************

}

# 切片作为函数参数

func main () {
  s:=[]int{1,2,3,4,5}
  test(s)
  fmt.printIn(s)  //这时候输出的还是[1,2,3,4,5]
}
func test(s []int){
  s=append(s,123,2)
}

// 解析：虽然切片作为函数函数是地址传递，但是切片的append操作会
        改变s的地址，所以不会影响原始地址。因此并不会改变main
        函数中s的值

# map 的删除

func mian () {
  m:=map[int]string{101:"刘备",102:"曹操",103:"孙权"}
  //删除不存在的key值也不会报错，没有返回值
  delete(m,101)
  fmt.printIn(m)    // [102:"曹操",103:"孙权"]
}

# map 作为函数参数

func main () {
  m := make(map[int]string,1)
  m[101] = "孙悟空"
  test(m)
  fmt.printIn(m)  // [101:"杨二郎",102:"天蓬元帅"]
  // 因为切片作为函数参数是地址传递，所以会改变主调函数的值
}
func test(m map[int]string){
  m[101] = "杨二郎"
  m[102] =  "天蓬元帅"
}
