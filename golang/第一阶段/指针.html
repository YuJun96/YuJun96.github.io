# 指针的定义

func main () {
  var p *int
  //定义一个整型指针
  var a int = 10
  p = &a           // 把a的地址赋值给p
  fmt.PrintIn(p)  //打印的是一个地址
  fmt.PrintIn(a)  // 10
  fmt.PrintIn(*p)  // 10
}

# 指针变量的 new 操作

func main () {
  var p *int
  fmt.printIn(p)  //<nil>

  p = new(int) //为指针变量创建一块内存空间，在堆区创建地址
  fmt.printIn(p) //0xff12344
  fmt.printIn(*p)  //0
}

# 通过 new 关键字创建切片

func main () {
  var p *[]int
  // 这时候不能给p直接赋值，因为它默认是一个空的切片指针
  // 需要通过new关键字给它在内存中开辟空间
  p = new([]int)
  *p = append(*p,1,2,3)
  fmt.printIn(*p)   //  [1,2,3]
}
