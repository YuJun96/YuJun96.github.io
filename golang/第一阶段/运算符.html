# 逻辑运算符

1.分为以下几类 || && ！= 2.一般&& 的优先级要高于 ||

func main() {
  var a:=10
  var b:=20
  var c:=30
  var d:=40
  fmt.printIn(a>b&&c>d)  //false  一句口诀：同真为真，否者为false
  fmt.printIn(a>b||c<d)  //true 一真为真，全假为假
}
