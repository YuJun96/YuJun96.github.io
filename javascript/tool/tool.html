# 函数节流

  1. 函数节流指的是某个函数在一定时间间隔内（例如 3 秒）只执行一次，在这 3 秒内 无视后来产生的函数调用请求，也不会延长时间间隔。3 秒间隔结束后第一次遇到新的函数调用会触发执行，然后在这新的 3 秒内依旧无视后来产生的函数调用请求，以此类推。

方法一:利用时间戳实现

function throttle(func, wait) {
  var context, args // 定义两个变量
  var time = 0 // 用于保存初始时间
  return function() {
    var nowTime = +new Date() // 拿到当前的时间,相当于new Date().getTime()
    context = this // 用于保存this
    args = arguments // 用于保存参数，防止当参数为时间对象e时，打印undefined
    if (nowTime - time > wait) {
      func.apply(context, args)
      time = nowTime
    }
  }
}

function throttle(func, wait) {
  var context, args // 定义两个变量
  var time = null // 用于记录定时器
  return function() {
    context = this
    args = arguments
    if (!timer) {
      timer = setTimeout(function() {
        timer = null
        func.apply(context, args)
      }, wait)
    }
  }
}

# 函数防抖

  1. 防抖，就是指触发事件后在 n 秒内函数只能执行一次，如果在 n 秒内又触发了事件，则会重新计算函数执行时间。

方法一: 非立即执行

function debounce(func, wait) {
  var args, context
  var timer
  return function() {
    context = this
    args = arguments
    if (timer) {
      clearTimeout(timer)
    }
    timer = setTimeout(() => {
      func.apply(context, args)
    }, wait)
  }
}

function debounce(func, wait) {

  var timer
  return function() {
    arg = arguments
    context = this
    if (timer) {
      clearTimeout(timer)
    }
    let callNow = !timer
    timer = setTimeout(() => {
      timer = null
    }, wait)
    if (callNow) func.apply(context, args)
  }
}

# 时间格式化

// 将 Date 类型转化为 String fmt:yyyy-MM-dd hh:mm:ss 或者 yyyy-MM-dd
function formatDate(date, fmt) {
  if (/(y+)/.test(fmt)) {
    fmt = fmt.replace(
      RegExp.$1,
      (date.getFullYear() + '').substr(4 - RegExp.$1.length)
    )
  }
  let o = {
    'M+': date.getMonth() + 1,
    'd+': date.getDate(),
    'h+': date.getHours(),
    'm+': date.getMinutes(),
    's+': date.getSeconds()
  }
  for (let k in o) {
    if (new RegExp(`(${k})`).test(fmt)) {
      let str = o[k] + ''
      fmt = fmt.replace(
        RegExp.$1,
        RegExp.$1.length === 1 ? str : ('00' + str).substr(str.length)
      )
    }
  }
  return fmt
}

//中国标准时间转换为"yyyy/MM/dd hh:mm "
function formatDateTime(date) {
  var y = date.getFullYear()
  var m = date.getMonth() + 1
  m = m < 10 ? '0' + m : m
  var d = date.getDate()
  d = d < 10 ? '0' + d : d
  var h = date.getHours()
  h = h < 10 ? '0' + h : h
  var minute = date.getMinutes()
  minute = minute < 10 ? '0' + minute : minute
  return y + '/' + m + '/' + d + ' ' + h + ':' + minute
}
